(defparameter increments 0)
(defparameter last -1)

(with-open-file (f "./input_day1.txt" :direction :input)
  (loop for num = (read f nil)
        while num
        if (and (> num last) (/= last -1))
          do (incf increments)
        always (setq last num)))
(format t "answer: ~a" increments)
